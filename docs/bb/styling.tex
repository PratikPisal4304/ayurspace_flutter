% --- Chapter title formatting ---
\titleformat{\chapter}[display]
  {\vspace{1cm}\normalfont\Large\bfseries\centering}
  {\chaptertitlename\ \thechapter}{20pt}{\vspace{0pt}}
\titlespacing*{\chapter}{0pt}{-50pt}{20pt}

% --- Fancy header/footer (uses metadata commands) ---
\pagestyle{fancy}
\renewcommand{\headrulewidth}{1pt}
\renewcommand{\footrulewidth}{1pt}
\lhead{}
\rhead{\footnotesize{\ProjectShortName}}
\cfoot{}
\lfoot{\footnotesize Pillai HOC College of Engineering and Technology, Rasayani (Autonomous)}
\rfoot{\thepage}

% Redefine 'plain' to match fancy so \chapter* pages also get headers/footers
\fancypagestyle{plain}{%
    \fancyhf{}%
    \renewcommand{\headrulewidth}{1pt}%
    \renewcommand{\footrulewidth}{1pt}%
    \lhead{}%
    \rhead{\footnotesize{\ProjectShortName}}%
    \cfoot{}%
    \lfoot{\footnotesize Pillai HOC College of Engineering and Technology, Rasayani (Autonomous)}%
    \rfoot{\thepage}%
}

% --- Abbreviations list environment ---
\newcommand{\abbrlabel}[1]{\makebox[3cm][l]{\textbf{#1}\ \dotfill}}
\newenvironment{abbreviations}{\begin{list}{}{\renewcommand{\makelabel}{\abbrlabel}}}{\end{list}}

% --- List of Figures / Tables formatting ---
\setlength{\cftfigindent}{0pt}
\setlength{\cftfignumwidth}{3.2cm}
\renewcommand{\cftfigpresnum}{Figure }
\setlength{\cfttabindent}{0pt}
\setlength{\cfttabnumwidth}{3.2cm}
\renewcommand{\cfttabpresnum}{Table }

% Restore centered bold titles for LOF, LOT, and TOC
\renewcommand{\cftloftitlefont}{\hfill\Large\bfseries}
\renewcommand{\cftafterloftitle}{\hfill}
\renewcommand{\cftlottitlefont}{\hfill\Large\bfseries}
\renewcommand{\cftafterlottitle}{\hfill}
\renewcommand{\cfttoctitlefont}{\hfill\Large\bfseries}
\renewcommand{\cftaftertoctitle}{\hfill}

% Reduce top padding above LOF, LOT, TOC titles
\setlength{\cftbeforeloftitleskip}{0pt}
\setlength{\cftbeforelottitleskip}{0pt}
\setlength{\cftbeforetoctitleskip}{0pt}

% --- Chapter cover page macro ---
% Usage: \ChapterCover{Title}
% \thechapter is used automatically — no need to pass the number.
% Place \clearpage BEFORE calling this (from chapter 2 onwards) to flush floats.
\newcommand{\ChapterCover}[1]{%
  \refstepcounter{chapter}%
  \addcontentsline{toc}{chapter}{\protect\numberline{\thechapter}#1}%
  \vspace*{\fill}%
  \begin{center}%
    \textbf{\Huge{Chapter \thechapter}}\\%
    \vspace{5mm}%
    \textbf{\Huge{#1}}%
  \end{center}%
  \vspace*{\fill}%
  \newpage%
}
