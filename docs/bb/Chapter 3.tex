\section*{3.1 \quad Software and Hardware Requirements}
\addcontentsline{toc}{section}{3.1 Software and Hardware Requirements}
\justify

\begin{enumerate}[label={\Roman*.}]
    \item \textbf{Software Requirements:}
    \begin{itemize}
        \item Framework \hspace{20mm}: Flutter (Dart SDK \^{}3.5.0)
        \item State Management \hspace{6mm}: Riverpod
        \item Cloud Infrastructure \hspace{3mm}: Firebase (Authentication, Firestore)
        \item APIs \hspace{28mm}: Google Gemini API (\texttt{gemini-2.5-flash}), Plant.id API v3
        \item Navigation \hspace{18mm}: GoRouter
        \item Code Editor \hspace{15mm}: Visual Studio Code / Android Studio
        \item Version Control \hspace{9mm}: Git
    \end{itemize}

    \item \textbf{Hardware Requirements (For Development):}
    \begin{itemize}
        \item Processor \hspace{19mm}: Intel i5 / Apple M1 or higher
        \item RAM \hspace{26mm}: 8 GB (16 GB Recommended)
        \item Storage \hspace{22mm}: 256 GB SSD or higher
        \item Mobile Device \hspace{12mm}: Android 9+ or iOS 14+ for testing
    \end{itemize}

    \item \textbf{Hardware Requirements (For End Users):}
    \begin{itemize}
        \item Smartphone with rear camera (8 MP or higher)
        \item Operating System: Android 9.0 (Pie) or iOS 14 and above
        \item RAM: 2 GB minimum
        \item Active internet connection (4G LTE or Wi-Fi)
    \end{itemize}
\end{enumerate}
