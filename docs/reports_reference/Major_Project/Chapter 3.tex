\subsection{Software and Hardware Requirements}

\vspace{5pt}

\noindent Here we will discuss everything we will need in order to execute the project. Below we list the necessary hardware and software requirements. 

\vspace{10pt}

% SUBSECTION 3.1.1
\subsubsection{Software Requirements}

\begin{itemize}

\item Code Editor: Visual Studio Code // Version 1.93.0

\item Backend Framework: Flask // Version 3.0.3

\item Programming Languages:
    \begin{itemize}
        \item Development - Python 3.12
        \item Frontend - HTML5, CSS3, JavaScript
    \end{itemize}

\item Database: SQLite3

\item Package Manager: pip // Version 24.0

\item Authentication: Flask-Login with Bcrypt password hashing

\item API Service:
    \begin{itemize}
       
        \item Google Speech-to-Text API (for voice input)
        \item Text-to-Speech Engine (for audio output)
    \end{itemize}

\item Environment Manager: python-dotenv

\end{itemize}
\vspace{10pt}

% SUBSECTION 3.1.2
\subsubsection{Hardware Requirements}

\begin{itemize}

\item Operating System: Microsoft Windows 10 / 11

\item RAM: Minimum 8 GB

\item Storage: 512 GB SSD

\item Processor: Intel i5 / AMD Ryzen 5 or higher

\item Web Browser: Google Chrome / Microsoft Edge

\end{itemize}
